// Create table
CREATE TABLE employees(
	employee_id integer,
	first_name varchar(20),
	last_name varchar(25) not null,
	email varchar(25) not null unique,
	phone_number varchar(20),
	hire_date date not null,
	salary numeric(8, 2),
	commission_pct numeric(2, 2),
	department_id integer
	);
